1    package ru.kpp.incidentmanager.controllers;
2    
3    import org.springframework.http.HttpStatus;
4    import org.springframework.web.bind.annotation.PathVariable;
5    import org.springframework.web.bind.annotation.RequestBody;
6    import org.springframework.web.bind.annotation.ResponseBody;
7    import org.springframework.web.bind.annotation.RestController;
8    import org.springframework.web.server.ResponseStatusException;
9    import ru.kpp.incidentmanager.entity.Incident;
10   import ru.kpp.incidentmanager.entity.Status;
11   import ru.kpp.incidentmanager.repositories.IncidentRepository;
12   
13   import java.util.NoSuchElementException;
14   
15   /**
16    * Класс отвечающий за основное взаимодействие с записями об инцидентах, построенные по принципу RESTful
17    * веб-службы CRUD (Create, Read, Update, Delete). Внешнее взаимодействие организовано в формате JSON.
18    *
19    * @author Ponkratov K.
20    */
21   
22   @RestController
23   public class IncidentsController {
24   
25       final
26       IncidentRepository incidentRepository;
27   
28   
29       public IncidentsController(IncidentRepository incidentRepository) {
30           this.incidentRepository = incidentRepository;
31       }
32   
33       /**
34        * Веб-метод добавления записи об инцидентев базу данных
35        *
36        * @param incident - передается экземпляр класса Incident для добавления в базу данных
37        * @return экземпляр класса Incident, записанный в базу данных
38        */
39   
40       public Incident addIncident(@RequestBody Incident incident) {
41           return incidentRepository.save(incident);
42       }
43   
44       /**
45        * Веб-метод получения записи об инциденте из базы данных по ID номеру записи
46        *
47        * @param id - номер запрашиваемой записи
48        * @return - возвращается экземпляр класса Incident
49        */
50   
51       public Incident getIncident(@PathVariable int id) {
52           try {
53               return incidentRepository.findById(id);
54           } catch (NoSuchElementException e) {
55               throw new ResponseStatusException(HttpStatus.NOT_FOUND, "Invalid incident number", e);
56           }
57       }
58   
59       /**
60        * Веб-метод получения всех записей об инцидентах из базы данных
61        *
62        * @return - возвращается массив экземпляров класса Incident
63        */
64   
65       public Iterable<Incident> getAllIncidents() {
66           return incidentRepository.findAll();
67       }
68   
69   
70       /**
71        * Веб-метод обновления записи об инциденте в базе данных
72        *
73        * @param incident - экземпляр класса Incident в формате JSON передаваемый для обновления
74        * @param id       - номер обновляемой записи
75        */
76   
77       public Incident updateIncident(@RequestBody Incident incident, @PathVariable int id) {
78           Incident incidentById = incidentRepository.findById(id);
79           incidentById = incident;
80           return incidentRepository.save(incidentById);
81       }
82   
83       /**
84        * Веб-метод удаления записи об инциденте в базе даннх
85        *
86        * @param id номер записи об инциденте для удаления
87        */
88   
89       public void deleteIncident(@PathVariable int id) {
90           incidentRepository.deleteById(id);
91       }
92   
93       /**
94        * Веб-метод получения всех записей об инцидентах из базы данных со статусом, отличным от статуса "Решен"
95        *
96        * @return - возвращает коллекцию экземпляров класса Incident
97        */
98   
99       public @ResponseBody
100      Iterable<Incident> getIncidentsInService(Status status) {
101          return incidentRepository.findAllByStatusIsNot(status);
102      }
103  
104  }
