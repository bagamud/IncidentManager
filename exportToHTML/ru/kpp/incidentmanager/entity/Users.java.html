3    import javax.persistence.*;
4    import java.util.Objects;

6    
7    /**
8     * Класс сущности пользователей, соответствет классу серверной части приложения, используется для формирования сведений
9     * о пользователе для аутентификации запросов к ресурсам, а также при формировании справочника, используемого
10    * веб-формами
11    */
12   
13   @Entity
14   @Table(schema = "inm")
15   public class Users {
16       private int id;
17       private String username;
18       private String password;
19       private String name;
20       private Roles role;
21       private String contacts;
22       private String email;
23   
24       @Id
25       @Column(name = "ID")
26       public int getId() {
27           return id;
28       }
29   
30       public void setId(int id) {
31           this.id = id;
32       }
33   
34       @Basic
35       @Column(name = "username")
36       public String getUsername() {
37           return username;
38       }
39   
40       public void setUsername(String username) {
41           this.username = username;
42       }
43   
44       @Basic
45       @Column(name = "PASSWORD")
46       public String getPassword() {
47           return password;
48       }
49   
50       public void setPassword(String password) {
51           this.password = password;
52       }
53   
54       @Basic
55       @Column(name = "NAME")
56       public String getName() {
57           return name;
58       }
59   
60       public void setName(String name) {
61           this.name = name;
62       }
63   
64       @ManyToOne(targetEntity = Roles.class, fetch = FetchType.EAGER)
65       @JoinColumn(name = "role")
66       public Roles getRole() {
67           return role;
68       }
69   
70       public void setRole(Roles role) {
71           this.role = role;
72       }
73   
74       @Basic
75       @Column(name = "CONTACTS")
76       public String getContacts() {
77           return contacts;
78       }
79   
80       public void setContacts(String contacts) {
81           this.contacts = contacts;
82       }
83   
84       @Basic
85       @Column(name = "EMAIL")
86       public String getEmail() {
87           return email;
88       }
89   
90       public void setEmail(String email) {
91           this.email = email;
92       }
93   
94       @Override
95       public boolean equals(Object o) {
96           if (this == o) return true;
97           if (o == null || getClass() != o.getClass()) return false;
98           Users users = (Users) o;
99           return id == users.id &&
100                  role == users.role &&
101                  Objects.equals(username, users.username) &&
102                  Objects.equals(password, users.password) &&
103                  Objects.equals(name, users.name) &&
104                  Objects.equals(contacts, users.contacts) &&
105                  Objects.equals(email, users.email);
106      }
107  
108      @Override
109      public int hashCode() {
110          return Objects.hash(id, username, password, name, role, contacts, email);
111      }
112  }
