101      @GetMapping(path = "/manager/add")
102      public String addIncident(@Valid IncidentForm incidentForm, /*BindingResult bindingResult,*/ Model model) {
103          getVocabulary(model);
104  //        if (bindingResult.hasErrors()) {
105  //            model.addAttribute("errors", bindingResult.getFieldErrors());
106  //        }
107          Incident incident = incidentTransForm.convert(incidentForm);

112      /**
113       * Метод контроллера реализующий обновление записи об инциденте из веб-формы в баз данных, возвращает обновленную
114       * запись из базы данных и записывает в аттрибуты для отображения в веб-форме

116       * @param incidentForm сведения об инциденте переданные с веб-формы

123      public String updateIncident(@Valid @RequestBody IncidentForm incidentForm, @PathVariable int id, Model model) {
124          getVocabulary(model);
125          Incident incident = incidentTransForm.convert(incidentForm);
126          model.addAttribute("incident", incidentTransForm.convert(incidentsController.updateIncident(incident, id)));

130      /**
131       * Метод контроллера реализующий обновление в баз данных статуса записи об инциденте на "Решен" и формирование даты
132       * решения заявки, возвращает обновленную
133       * запись из базы данных и записывает в аттрибуты для отображения в веб-форме
134       *
135       * @param incidentForm сведения об инциденте переданные с веб-формы
136       * @param id           идентификатор записи об инциденте
137       * @param model        интерфейс определяющий аттрибуты
138       * @return возвращяет путь к странице
139       */
140  
141      @GetMapping(path = "/manager/done/{id}")
142      public String fixIncident(@Valid IncidentForm incidentForm, @PathVariable int id, Model model) throws ParseException {
143          getVocabulary(model);
144          int influence = Integer.parseInt((String) Objects.requireNonNull(model.getAttribute("influence")));
145          int urgency = Integer.parseInt((String) Objects.requireNonNull(model.getAttribute("urgency")));
146          incidentForm.setPriority(urgency - 1 + influence);
147          incidentForm.setStatus(3);
148          incidentForm.setClosedate(new Timestamp(new Date().getTime()));
149          Incident incident = incidentTransForm.convert(incidentForm);
150          model.addAttribute("incident", incidentTransForm.convert(incidentsController.updateIncident(incident, id)));
151          return "manager";
152      }
