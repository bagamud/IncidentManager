1    package ru.kpp.incidentmanager.form;
2    
3    import org.springframework.stereotype.Service;
4    import ru.kpp.incidentmanager.entity.Incident;
5    import ru.kpp.incidentmanager.repositories.*;
6    
7    /**
8     * Класс предназначен для конвертирования экземпляра класса {@link Incident} в экземпляр класса {@link IncidentForm}
9     * для валидации и отображения пользователю в веб-форме manager.jsp и обратно, для взаимодействия введенной пользователем
10    * информации об инциденте с методами приложения
11    */
12   
13   @Service
14   public class IncidentTransForm {
15   
16       /**
17        * Объявление экземпляра клиента справочников
18        */
19   
20       final CategoryRepository categoryRepository;
21       final DepartmentRepository departmentRepository;
22       final PriorityRepository priorityRepository;
23       final StatusRepository statusRepository;
24       final UsersRepository usersRepository;
25   
26       public IncidentTransForm(CategoryRepository categoryRepository, DepartmentRepository departmentRepository,
27                                PriorityRepository priorityRepository, StatusRepository statusRepository,
28                                UsersRepository usersRepository) {
29           this.categoryRepository = categoryRepository;
30           this.departmentRepository = departmentRepository;
31           this.priorityRepository = priorityRepository;
32           this.statusRepository = statusRepository;
33           this.usersRepository = usersRepository;
34       }
35   
36       /**
37        * Метод конвертирования экземпляра класса представления {@link IncidentForm} в класс сущности {@link Incident}
38        *
39        * @param incidentForm экземпляр класса {@link IncidentForm}
40        * @return экземпляр класса {@link Incident}
41        */
42   
43       public Incident convert(IncidentForm incidentForm) {
44   
45           Incident incident = new Incident();
46   
47           incident.setId(incidentForm.getId());
48           incident.setDate(incidentForm.getDate());
49           incident.setCategory(categoryRepository.findById(incidentForm.getCategory()));
50           incident.setPriority(priorityRepository.findById(incidentForm.getPriority()));
51           incident.setRequesterdepartment(departmentRepository.findById(incidentForm.getRequesterdepartment()));
52           incident.setRequester(incidentForm.getRequester());
53           incident.setRequestercontacts(incidentForm.getRequestercontacts());
54           incident.setIpaddress(incidentForm.getIpaddress());
55           incident.setDuration(incidentForm.getDuration());
56           incident.setDescription(incidentForm.getDescription());
57           incident.setEngineer(usersRepository.findById(incidentForm.getEngineer()));
58           incident.setOperator(usersRepository.findById(incidentForm.getOperator()));
59           incident.setStatus(statusRepository.findById(incidentForm.getStatus()));
60           incident.setClosedate(incidentForm.getClosedate());
61           incident.setJournal(incidentForm.getJournal());
62   
63           return incident;
64       }
65   
66       /**
67        * Метод конвертирования экземпляра класса сущности {@link Incident} в класс представления {@link IncidentForm}
68        *
69        * @param incident экземпляр класса {@link Incident}
70        * @return экземпляр класса {@link IncidentForm}
71        */
72   
73       public IncidentForm convert(Incident incident) {
74   
75           IncidentForm incidentForm = new IncidentForm();
76   
77           incidentForm.setId(incident.getId());
78           incidentForm.setDate(incident.getDate());
79           incidentForm.setCategory(incident.getCategory().getId());
80           incidentForm.setCategoryTitle(incident.getCategory().getTitle());
81           if (incident.getPriority() != null) {
82               incidentForm.setPriority(incident.getPriority().getId());
83               incidentForm.setPriorityDescription(incident.getPriority().getDescription());
84           }
85           incidentForm.setRequesterdepartment(incident.getRequesterdepartment().getId());
86           incidentForm.setRequesterdepartmentTitle(incident.getRequesterdepartment().getTitle());
87           incidentForm.setRequester(incident.getRequester());
88           incidentForm.setRequestercontacts(incident.getRequestercontacts());
89           incidentForm.setIpaddress(incident.getIpaddress());
90           if (incident.getDuration() != null) incidentForm.setDuration(incident.getDuration());
91           incidentForm.setDescription(incident.getDescription());
92           incidentForm.setEngineer(incident.getEngineer().getId());
93           incidentForm.setEngineerName(incident.getEngineer().getName());
94           incidentForm.setOperator(incident.getOperator().getId());
95           incidentForm.setOperatorName(incident.getOperator().getName());
96           incidentForm.setStatus(incident.getStatus().getId());
97           incidentForm.setStatusTitle(incident.getStatus().getTitle());
98           incidentForm.setClosedate(incident.getClosedate());
99           incidentForm.setJournal(incident.getJournal());
100  
101          return incidentForm;
102      }
103  }
