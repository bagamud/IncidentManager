1    package ru.kpp.incidentmanager.securingweb;
2    
3    import org.springframework.context.annotation.Bean;
4    import org.springframework.context.annotation.Configuration;
5    import org.springframework.security.config.annotation.authentication.builders.AuthenticationManagerBuilder;
6    import org.springframework.security.config.annotation.web.builders.HttpSecurity;
7    import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
8    import org.springframework.security.config.annotation.web.configuration.WebSecurityConfigurerAdapter;
9    import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;
10   import org.springframework.security.crypto.password.PasswordEncoder;
11   
12   /**
13    * В данном классе определяется конфигурация веб-безопасности приложения в части, определения порядка доступа к ресурсам
14    * приложения и рапределения доступа к ресурсам авторизованных и неавторизованных пользователей,
15    * определяения службы кодирования паролей и определяения конфигурации службы аутентификации
16    */
17   
18   @Configuration
19   @EnableWebSecurity
20   public class WebSecurityConfig extends WebSecurityConfigurerAdapter {
21   
22       /**
23        * Объявление сервиса, формирующего сведения о пользователе
24        */
25       final
26       UsersService usersService;
27   
28       public WebSecurityConfig(UsersService usersService) {
29           this.usersService = usersService;
30       }
31   
32       /**
33        * Метод определяет порядок защиты доступа к ресурсам приложение по URL-путям
34        *
35        * @param http элемент конфигурации безопасности отвечающий за HTTP-запросы
36        * @throws Exception общие исключения в процессе доступа к ресурсам
37        */
38   
39       @Override
40       protected void configure(HttpSecurity http) throws Exception {
41           http
42                   .authorizeRequests()
43                   .antMatchers("/dashboard", "/manager").authenticated()
44                   .and()
45                   .httpBasic()
46                   .realmName("Incident Manager")
47                   .and()
48                   .formLogin()
49                   //                .loginPage("/auth")
50                   .permitAll().defaultSuccessUrl("/dashboard")
51                   .and()
52                   .logout()
53                   .permitAll()
54                   .logoutSuccessUrl("/");
55       }
56   
57       /**
58        * Метод определяет службу, кодирующую пароли
59        *
60        * @return экземпляр службы кодирования
61        */
62   
63       @Bean
64       public PasswordEncoder passwordEncoder() {
65           return new BCryptPasswordEncoder();
66       }
67   
68       /**
69        * Метод определяющий конфигурацию службы аутентификации пользователей
70        *
71        * @param builder экземпляр менеджера аутентификации
72        * @throws Exception проброс общих исключений формирования менеджера
73        */
74   
75       @Override
76       protected void configure(AuthenticationManagerBuilder builder) throws Exception {
77           builder.userDetailsService(usersService).passwordEncoder(passwordEncoder());
78       }
79   }
