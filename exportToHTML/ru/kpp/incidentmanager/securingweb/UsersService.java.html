1    package ru.kpp.incidentmanager.securingweb;
2    
3    import org.springframework.security.core.authority.SimpleGrantedAuthority;
4    import org.springframework.security.core.userdetails.User;
5    import org.springframework.security.core.userdetails.UserDetails;
6    import org.springframework.security.core.userdetails.UserDetailsService;
7    import org.springframework.security.core.userdetails.UsernameNotFoundException;
8    import org.springframework.stereotype.Service;
9    import ru.kpp.incidentmanager.entity.Users;
10   import ru.kpp.incidentmanager.repositories.UsersRepository;
11   
12   import java.util.Collections;
13   import java.util.List;
14   
15   /**
16    * Класс формирующий сведения о пользователе, проходящем авторизацию, посредством запроса через клиент
17    * к записям о зарегистрированных пользователях, содержащихся в базе данных
18    */
19   
20   @Service
21   public class UsersService implements UserDetailsService {
22   
23       /**
24        * Объявление клиента, формирующего HTTP-запросы к серверной части приложения. Используется для получения
25        * сведений о пользователе
26        */
27   
28       final UsersRepository usersRepository;
29   
30       public UsersService(UsersRepository usersRepository) {
31           this.usersRepository = usersRepository;
32       }
33   
34       /**
35        * Метод запрашивающий экземпляр класса {@link Users}, по совпадению имени пользователя,
36        * введенного в форму авторизации
37        *
38        * @param username имя пользователя, передается из формы авторизации
39        * @return экземпляр класса {@link UserDetails}, содержащий сведения о пользователе для дальнейшей аутентификации
40        * @throws UsernameNotFoundException исключение, если по имени пользователя, нет соответствующей записи
41        *                                   в базе данных зарегистрированных пользователей
42        */
43   
44       @Override
45       public UserDetails loadUserByUsername(String username) throws UsernameNotFoundException {
46   
47           /* 
48             Создание экземпляра класса {@link Users}, поиск по имени пользователя, и присвоение значений экземпляра 
49             записи о пользователе 
50            */
51   
52           Users user = usersRepository.findByUsername(username);
53   
54           /* 
55             Проверка на null значение переменной user 
56            */
57   
58           if (user == null) {
59               throw new UsernameNotFoundException("Пользователь не найден");
60           }
61   
62           /* 
63           Создание единичного массива из значения о роли пользователя 
64            */
65   
66           List<SimpleGrantedAuthority> authorities = Collections.singletonList(new SimpleGrantedAuthority(user.getRole().getName()));
67           return new User(user.getUsername(), user.getPassword(), authorities);
68       }
69   }
